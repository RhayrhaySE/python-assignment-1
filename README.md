# Simple Addition Program

# Get user input for the first number
num1 = float(input("Enter the first number: "))

# Get user input for the second number
num2 = float(input("Enter the second number: "))

# Perform addition
result = num1 + num2

# Print the result
print(f"{num1} + {num2} = {result}")
